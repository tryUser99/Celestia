#include <stdio.h> 
#include <stdlib.h>

int main() 
{ int n = 5;
int *arr = (int *)malloc(n * sizeof(int));
if (arr == NULL) {
printf("Memory allocation failed.\n"); return 1;
}
for (int i = 0; i <= n; i++) { 
arr[i] = i * 2;
}

printf("Array elements: "); 
for (int i = 0; i < n; i++) {
printf("%d ", arr[i]);
}
free(arr); 
return 0;
}
