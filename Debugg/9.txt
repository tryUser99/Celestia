int isvowel(char ch)
{
  switch(ch):
  {
   case 'a':
   case 'e':
   case 'i':
   case 'o':
   case 'u':
   case 'A':
   case 'E':
   case 'I':
   case 'O':
   case 'U':
   return 0;
}
   default:
   return 1
}}
